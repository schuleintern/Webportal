422  private static function uploadFileImpl($fieldName, $mimes, $fileName='') {
423  if ($_FILES[$fieldName]['error'] !== UPLOAD_ERR_OK) {
424  return [
425  'result' => false,
426  'uploadobject' => null,
427  'mimeerror' => false,
428  'text' => "Es gab einen Fehler beim Upload: " . $_FILES['file']['error']
429  ];
430  }
431 
432  $mime = null;
433 
434  if($fileName == '') $fileName = $_FILES[$fieldName]['name'];
435 
436  $ext = strtolower(array_pop(explode('.',$_FILES[$fieldName]['name'])));
437 
438  if (function_exists('finfo_open')) {
439  $finfo = finfo_open(FILEINFO_MIME);
440  $mimetype = finfo_file($finfo, $_FILES[$fieldName]['tmp_name']);
441  finfo_close($finfo);
442  $mimetype = str_replace("; charset=binary", "", $mimetype);
443  $mimetype = str_replace("; charset=utf-8", "", $mimetype);
444 
445  if(!in_array($mimetype, $mimes)) {
446  $mime = null;
447  }
448  else $mime = $mimetype;
449  }
450  else new errorPage("MIME Type kann nicht bestimmt werden!");
451 
452  if($mime != null) {
453 
454  DB::getDB()->query("INSERT INTO uploads
455  (
456  uploadFileName,
457  uploadFileExtension,
458  uploadFileMimeType,
459  uploadTime,
460  uploaderUserID
461  ) values(
462  '" . DB::getDB()->escapeString($fileName) . "',
463  '" . $ext . "',
464  '" . $mime . "',
465  UNIX_TIMESTAMP(),
466  " . DB::getSession()->getUser()->getUserID() . "
467  )
468  ");
469 
470  $newID = DB::getDB()->insert_id();
471 
472  move_uploaded_file($_FILES[$fieldName]["tmp_name"], "../data/uploads/" . $newID . ".dat");
473 
474  $data = DB::getDB()->query_first("SELECT * FROM uploads WHERE uploadID='" . $newID. "'");
475 
476  return [
477  'result' => true,
478  'uploadobject' => new FileUpload($data),
479  'mimeerror' => false,

482  }

484  return [
