244  public function buildPropertyBag()
245  {
246  $propertyBag = new PropertyBag();
247 
248  // mandatory information
249  $propertyBag->set('UID', $this->uniqueId);
250 
251  $propertyBag->add(new DateTimeProperty('DTSTART', $this->dtStart, $this->noTime, $this->useTimezone, $this->useUtc, $this->timezoneString));
252  $propertyBag->set('SEQUENCE', $this->sequence);
253  $propertyBag->set('TRANSP', $this->transparency);
254 
255  if ($this->status) {
256  $propertyBag->set('STATUS', $this->status);
257  }
258 
259  // An event can have a 'dtend' or 'duration', but not both.
260  if ($this->dtEnd !== null) {
261  $dtEnd = clone $this->dtEnd;
262  if ($this->noTime === true) {
263  $dtEnd = $dtEnd->add(new \DateInterval('P1D'));
264  }
265  $propertyBag->add(new DateTimeProperty('DTEND', $dtEnd, $this->noTime, $this->useTimezone, $this->useUtc, $this->timezoneString));
266  } elseif ($this->duration !== null) {
267  $propertyBag->set('DURATION', $this->duration->format('P%dDT%hH%iM%sS'));
268  }
269 
270  // optional information
271  if (null != $this->url) {
272  $propertyBag->set('URL', $this->url);
273  }
274 
275  if (null != $this->location) {
276  $propertyBag->set('LOCATION', $this->location);
277 
278  if (null != $this->locationGeo) {
279  $propertyBag->add(
280  new Property(
281  'X-APPLE-STRUCTURED-LOCATION',
282  new RawStringValue('geo:' . $this->locationGeo->getGeoLocationAsString(',')),
283  [
284  'VALUE' => 'URI',
285  'X-ADDRESS' => $this->location,
286  'X-APPLE-RADIUS' => 49,
287  'X-TITLE' => $this->locationTitle,
288  ]
289  )
290  );
291  }
292  }
293 
294  if (null != $this->locationGeo) {
295  $propertyBag->add($this->locationGeo);
296  }
297 
298  if (null != $this->summary) {
299  $propertyBag->set('SUMMARY', $this->summary);
300  }
301 
302  if (null != $this->attendees) {
303  $propertyBag->add($this->attendees);
304  }
305 
306  $propertyBag->set('CLASS', $this->isPrivate ? 'PRIVATE' : 'PUBLIC');
307 
308  if (null != $this->description) {
309  $propertyBag->set('DESCRIPTION', $this->description);
310  }
311 
312  if (null != $this->descriptionHTML) {
313  $propertyBag->add(
314  new Property(
315  'X-ALT-DESC',
316  $this->descriptionHTML,
317  [
318  'FMTTYPE' => 'text/html',
319  ]
320  )
321  );
322  }
323 
324  if (null != $this->recurrenceRule) {
325  $propertyBag->set('RRULE', $this->recurrenceRule);
326  }
327 
328  foreach ($this->recurrenceRules as $recurrenceRule) {
329  $propertyBag->set('RRULE', $recurrenceRule);
330  }
331 
332  if (null != $this->recurrenceId) {
333  $this->recurrenceId->applyTimeSettings($this->noTime, $this->useTimezone, $this->useUtc, $this->timezoneString);
334  $propertyBag->add($this->recurrenceId);
335  }
336 
337  if (!empty($this->exDates)) {
338  $propertyBag->add(new DateTimesProperty('EXDATE', $this->exDates, $this->noTime, $this->useTimezone, $this->useUtc, $this->timezoneString));
339  }
340 
341  if ($this->cancelled) {
342  $propertyBag->set('STATUS', 'CANCELLED');
343  }
344 
345  if (null != $this->organizer) {
346  $propertyBag->add($this->organizer);
347  }
348 
349  if ($this->noTime) {
350  $propertyBag->set('X-MICROSOFT-CDO-ALLDAYEVENT', 'TRUE');
351  }
352 
353  if (null != $this->categories) {
354  $propertyBag->set('CATEGORIES', $this->categories);
355  }
356 
357  $propertyBag->add(
358  new DateTimeProperty('DTSTAMP', $this->dtStamp ?: new \DateTimeImmutable(), false, false, true)
359  );
360 
361  if ($this->created) {
362  $propertyBag->add(new DateTimeProperty('CREATED', $this->created, false, false, true));
363  }
364 
365  if ($this->modified) {
366  $propertyBag->add(new DateTimeProperty('LAST-MODIFIED', $this->modified, false, false, true));
367  }
368 
369  return $propertyBag;
370  }
