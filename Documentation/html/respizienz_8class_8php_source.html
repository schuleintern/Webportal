376  private function tofloat($num) {
377  $dotPos = strrpos($num, '.');
378  $commaPos = strrpos($num, ',');
379  $sep = (($dotPos > $commaPos) && $dotPos) ? $dotPos :
380  ((($commaPos > $dotPos) && $commaPos) ? $commaPos : false);
381 
382  if (!$sep) {
383  return floatval(preg_replace("/[^0-9]/", "", $num));
384  }
385 
386  return floatval(
387  preg_replace("/[^0-9]/", "", substr($num, 0, $sep)) . '.' .
388  preg_replace("/[^0-9]/", "", substr($num, $sep+1, strlen($num)))
389  );
390  }
